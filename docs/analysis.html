Evidence-Based Insights into Industrial Engineering Programs

Key Takeaways

  • Hybrid Data Sourcing: Successfully integrated live API extraction (thestats) with static datasets (Kaggle) to construct a comprehensive timeline.
  • National Trends: Clear shifts identified in placement capacities and base rankings across Turkish universities.
  • Regional & Demand Correlations: Advanced modeling reveals strong relationships between a program’s geographical location, its demand-to-quota ratio, and its ultimate national success ranking.

1. Introduction to Analysis

As Industrial Engineering students, our objective is to apply data-driven decision-making to our own academic domain. In this section, we transition from raw admission data to actionable insights using a hybrid data engineering pipeline, utilizing tidyverse for data wrangling and ggplot2 for high-impact visualizations.

2. Hybrid Data Exploration & Cleaning

To execute our methodology of decoding the drivers of success, we extract deep numeric metrics (Rankings, Quotas, Preferences, Demand Ratios, and Top-Choice counts) alongside geographical data from our Kaggle integration.

Code 
library(tidyverse)
library(thestats)
library(DT) 

# ==========================================
# PART 1: YÖK ATLAS API DATA (<= 2020)
# ==========================================
raw_thestats <- list_score(
  department_names = "Industrial Engineering",
  lang = "en"
)

clean_thestats <- raw_thestats %>%
  filter(as.numeric(year) <= 2020) %>%
  filter(!str_detect(str_to_lower(department), "woodworking|fisheries|aquaculture|forest|design")) %>%
  mutate(
    University_Type = case_when(
      str_detect(str_to_lower(type), "devlet|state") ~ "State",
      str_detect(str_to_lower(type), "vak|foundation|private") ~ "Foundation",
      TRUE ~ "Other"
    )
  ) %>%
  filter(University_Type != "Other") %>%
  select(
    Year = year,
    University_Type,
    University_Name = university,
    Faculty_Name = faculty,
    Department_Name = department
  ) %>%
  mutate(Year = as.numeric(Year))

# ==========================================
# PART 2: KAGGLE DATASET (2021 - 2024) - KAPSAMLI VERİ ÇEKİMİ
# ==========================================
raw_kaggle <- read_csv("data/01_university_admissions_turkey_2019_2024.csv")

clean_kaggle <- raw_kaggle %>%
  filter(as.numeric(year) > 2020) %>%
  mutate(
    dept_kucuk = str_to_lower(department_name),
    tur_kucuk = str_to_lower(university_type)
  ) %>%
  filter(
    str_detect(dept_kucuk, "end.str") & str_detect(dept_kucuk, "m.hendis"),
    !str_detect(dept_kucuk, "orman|a.a.|tasar.m|su .r.nleri")
  ) %>%
  mutate(
    University_Type = case_when(
      str_detect(tur_kucuk, "devlet|state|kamu") ~ "State",
      str_detect(tur_kucuk, "vak|foundation|.zel|private") ~ "Foundation",
      TRUE ~ "Other"
    ),
    university_name = str_replace_all(university_name, "(?i).n.vers.tes.", "UNIVERSITY"),
    university_name = str_replace_all(university_name, "(?i)tekn.k", "TECHNICAL"),
    university_name = str_replace_all(university_name, "(?i)enst.t.s.", "INSTITUTE"),
    faculty_name = str_replace_all(faculty_name, "(?i)m.hend.sl.k", "Engineering"),
    faculty_name = str_replace_all(faculty_name, "(?i)fak.ltes.", "Faculty"),
    department_name = str_replace_all(department_name, "(?i)end.str.", "Industrial"),
    department_name = str_replace_all(department_name, "(?i)m.hend.sl...", "Engineering")
  ) %>%
  filter(University_Type != "Other") %>%
  # HEDEFLENEN YENİ VERİLER: Şehir, Talep Oranı, İlk Tercih Edenler
  select(
    Year = year,
    City = city,
    University_Type,
    University_Name = university_name,
    Faculty_Name = faculty_name,
    Department_Name = department_name,
    Rank = final_rank_012,
    Quota = total_quota,
    Preferences = total_preferences,
    Demand_Ratio = demand_per_quota,
    Top1_Pref = top_1_pref_count
  ) %>%
  mutate(
    Year = as.numeric(Year),
    Rank = as.numeric(Rank),
    Quota = as.numeric(Quota),
    Preferences = as.numeric(Preferences),
    Demand_Ratio = as.numeric(Demand_Ratio),
    Top1_Pref = as.numeric(Top1_Pref)
  )

# ==========================================
# PART 3: MASTER COMBINED DATASET
# ==========================================
ie_combined <- bind_rows(clean_thestats, clean_kaggle) %>%
  mutate(
    University_Name = str_to_title(str_squish(University_Name)),
    Faculty_Name = str_to_title(str_squish(Faculty_Name)),
    Department_Name = str_to_title(str_squish(Department_Name))
  ) %>%
  arrange(desc(Year), University_Name)

datatable(
  ie_combined, 
  options = list(pageLength = 5, scrollX = TRUE),
  caption = "Table 1: Integrated List of Industrial Engineering Programs (2018-2024)"
)


3. Exploratory Data Analysis (EDA): Structural Distribution

With our seamlessly integrated master dataset, we visualize the structural distribution of Industrial Engineering programs for the most recent academic year.

Code 
latest_year <- max(ie_combined$Year, na.rm = TRUE)

ie_combined %>%
  filter(Year == latest_year) %>%
  count(University_Type) %>%
  ggplot(aes(x = University_Type, y = n, fill = University_Type)) +
  geom_bar(stat = "identity", width = 0.5, show.legend = FALSE, alpha = 0.9) +
  scale_fill_manual(values = c("State" = "#38bdf8", "Foundation" = "#94a3b8")) +
  geom_text(aes(label = n), vjust = -0.5, fontface = "bold", color = "white", size = 5) +
  theme_minimal() +
  labs(
    title = paste("Distribution of IE Programs in", latest_year),
    subtitle = "Active Programs in State vs. Foundation Universities",
    x = "University Sector",
    y = "Number of Active Programs"
  ) +
  theme(
    plot.title = element_text(face = "bold", size = 16, color = "#38bdf8"),
    plot.subtitle = element_text(size = 12, color = "#cbd5e1", margin = margin(b = 15)),
    axis.text = element_text(color = "#f8fafc", size = 11),
    axis.title = element_text(face = "bold", color = "#f8fafc", size = 12),
    panel.grid.major.x = element_blank(),
    panel.grid.minor = element_blank(),
    panel.grid.major.y = element_line(color = "gray", linetype = "dotted")
  )

5. Regional Impact: Does Location Matter?

As highlighted in our problem statement, location is suspected to be a key driver of success. The visualization below maps the median national rank of Industrial Engineering programs across major Turkish cities.

Code 
ie_combined %>%
  filter(Year == latest_year, !is.na(City), !is.na(Rank)) %>%
  mutate(City = str_to_title(City)) %>%
  # Sadece program sayısı fazla olan büyük şehirleri alıyoruz
  group_by(City) %>%
  filter(n() >= 3) %>%
  ungroup() %>%
  # Median Rank'a göre sıralıyoruz (Düşük rakam en iyisi)
  ggplot(aes(x = reorder(City, -Rank, FUN = median, na.rm = TRUE), y = Rank, fill = City)) +
  geom_boxplot(alpha = 0.8, show.legend = FALSE, color = "#94a3b8") +
  scale_fill_viridis_d(option = "mako") +
  scale_y_reverse() +
  coord_flip() + # İsimler daha rahat okunsun diye yatay çevirdik
  theme_minimal() +
  labs(
    title = "Regional Drivers: City Influence on Rankings",
    subtitle = paste("Base Rank Distribution by City in", latest_year),
    x = "City",
    y = "National Base Rank (Lower is Better)"
  ) +
  theme(
    plot.title = element_text(face = "bold", size = 16, color = "#38bdf8"),
    plot.subtitle = element_text(size = 12, color = "#cbd5e1", margin = margin(b = 15)),
    axis.text = element_text(color = "#f8fafc", size = 11),

6. Advanced Correlation: Popularity Drivers vs. Success

To truly decode the drivers of success, we establish a multi-variable correlation model. We analyze how Demand Ratio (Total Preferences / Quota) and Top-Choice Exclusivity (Students who put the program as their #1 choice) correlate with the final National Rank.

Code 
ie_combined %>%
  filter(Year == latest_year, !is.na(Rank), !is.na(Demand_Ratio)) %>%
  ggplot(aes(x = Demand_Ratio, y = Rank, color = University_Type)) +
  geom_point(aes(size = Top1_Pref), alpha = 0.7) +
  geom_smooth(method = "lm", se = FALSE, linetype = "dashed", color = "#f8fafc") +
  scale_color_manual(values = c("State" = "#38bdf8", "Foundation" = "#f472b6")) +
  scale_y_reverse() + 
  theme_minimal() +
  labs(
    title = "Correlation: Demand Intensity vs. Academic Success",
    subtitle = "Higher demand per quota strongly correlates with better (lower) national ranks.",
    x = "Demand Ratio (Total Preferences / Quota)",
    y = "National Base Rank (Lower is Better)",
    size = "Top 1 Choices",
    color = "University Type",
    caption = "Data: Kaggle (Turkish University Admissions)"
  ) +
  theme(
    plot.title = element_text(face = "bold", size = 16, color = "#38bdf8"),
    plot.subtitle = element_text(size = 12, color = "#cbd5e1", margin = margin(b = 15)),
    axis.text = element_text(color = "#f8fafc", size = 11),
    axis.title = element_text(face = "bold", color = "#f8fafc", size = 12),
    legend.title = element_text(color = "#f8fafc", face = "bold"),
    legend.text = element_text(color = "#cbd5e1"),
    legend.position = "right",
    panel.grid.major = element_line(color = "#334155", linetype = "dotted"),
    panel.grid.minor = element_blank()
  )
